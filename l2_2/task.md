L2.2  
Объяснить порядок выполнения defer функций и итоговый вывод.  

```
package main

import "fmt"

func test() (x int) {
    // x - именованный возвращаемый параметр, при входе в функцию x = 0  
    // defer замыкает переменную x, а не её текущее значение  
    // при выполнении return сначала формируется возвращаемое значение,  
    // затем выполняется defer, который изменяет x  
    // так как x именованный, defer поменяет итоговый результат  

  defer func() {
    x++
  }()
  x = 1
  return // x = 2
}

func anotherTest() int {
    // в этом примере x int НЕ именованный возврат,  
    // при выполнении return x значение x копируется  
    // после этого выполняется defer-функция  
    // defer не поменяет возвращаемое значение   
    // функция вернет x = 1  
    
  var x int
  defer func() {
    x++
  }()
  x = 1
  return x // x = 1
}

func main() {
  fmt.Println(test())
  fmt.Println(anotherTest())
}

```
