Объяснить поведение срезов при передаче их в функцию.

```
package main

import (
  "fmt"
)

func main() {
  var s = []string{"1", "2", "3"}
  // При передаче слайсов в функцию происходит передача структуры, указывающая на общий массив  
  // поэтому изменения Слайса внутри функции изменит и оригинальный слайс 
  modifySlice(s)
  fmt.Println(s)
}

func modifySlice(i []string) {
    // замена i[0] = "3" len = 3 cap = 3, меняется и и оригинальный слайс по ссылке  {"3", "2", "3"}  
    // добавление "4" в конец массива cap удваивается и = 6 len = 4   
    // локальная переменная i начинает указывать на новый массив  
    // внутри функции переменная i теперь совсем другой слайс {"3", "2", "3", "4"}   
  i = append(i, "4")
    // замена в локальном слайсе i[1] = "5" len = 4 cap = 6, {"3", "5", "3"}
  i[1] = "5"
    // добавление "6" в конец массива len = 5, cap = 6  
   // {"3", "5", "3", "4", "6"}
  i = append(i, "6")
}
```
